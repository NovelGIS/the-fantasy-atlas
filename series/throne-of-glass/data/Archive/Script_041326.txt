                    fillColor: "gray",

                    pane: 'poiPane'

let allCharacterData;
var characterLayer;
var markers = L.markerClusterGroup({
    maxClusterRadius: 30,
    zoomToBoundsOnClick: false
});

function updateFilters() {
    showFiltered();
}

function showFiltered() {

    window.locationCounts = {};

    let selectedBook = document.getElementById("bookSelect").value;
    let selectedChapter = document.getElementById("chapterSelect").value;

    markers.clearLayers();

    characterLayer = L.geoJSON(allCharacterData, {

        filter: function (feature) {
            let matchBook = selectedBook ? feature.properties.Book === selectedBook : true;
            let matchChapter = selectedChapter ? feature.properties.Chapter == selectedChapter : true;
            return matchBook && matchChapter;
        },

        pointToLayer: function (feature, latlng) {

            const location = feature.properties.Location; // use a field that groups same-place points

            // Track how many points we've already placed at this location
            if (!window.locationCounts) window.locationCounts = {};
            if (!locationCounts[location]) locationCounts[location] = 0;

            const index = locationCounts[location]++;
            const total = 10; // approximate max points at a location (can be tweaked)

            // THIS is where angle lives
            const angle = (index / total) * Math.PI * 2;

            const distance = 0.00015; // tweak this for spacing

            const newLat = latlng.lat + Math.sin(angle) * distance;
            const newLng = latlng.lng + Math.cos(angle) * distance;

            return L.circleMarker([newLat, newLng], {
                radius: 2,
                fillColor: "white",
                color: "white",
                weight: 1,
                fillOpacity: 1,
            });
        },

        onEachFeature: function(feature, layer) {
            const name = feature.properties.Character;

            // Split label manually or format as needed
            const formatted = name.replace(" - ", "<br>");

            layer.bindTooltip(formatted, {
                permanent: true,
                direction: "center",
                className: "character-label",
            });

            
        }
    });

    map.on('zoomend', function() {
    markers.eachLayer(function(layer) {
        if (layer.getTooltip) {
            if (map.getZoom() >= 6) {
                layer.openTooltip();
            } else {
                layer.closeTooltip();
            }
        }
    });
});

    markers.addLayer(characterLayer);
    map.addLayer(markers);
}

fetch('Characters-Chapters_040426.geojson')
    .then(response => response.json())
    .then(data => {
        allCharacterData = data;

        console.log("Loaded features:", data.features.length);

        populateChapterDropdown();

        document.getElementById("bookSelect").value = "TOG";
        document.getElementById("chapterSelect").value = "1";

        showFiltered();
    });

    function populateChapterDropdown() {

        const chapterSelect = document.getElementById("chapterSelect");

        chapterSelect.innerHTML = '';

        let chapters = [...new Set(
            allCharacterData.features.map(f => f.properties.Chapter)
        )];

        chapters.sort((a, b) => a - b);

        chapters.forEach(ch => {
            let option = document.createElement("option");
            option.value = ch;
            option.text = "Chapter " + ch;
            chapterSelect.appendChild(option);
        });
    }
